Dynamic gradients

import * as format from 'conditionaljs';
// ...
const data = [];
for (let i = 0; i < 10; i++){
  data.push(Math.round(Math.random() * 10));
}
const colorStart = '#e96443';
const colorEnd = '#904e95';
const colors = format.generateTwoWayGradient(colorStart, colorEnd, data.length);
console.log(colors);

// colors -> ["#e0614b", "#d75f53", "#ce5d5b", "#c55b63", "#bc596c", "#b35674", "#aa547c", "#a15284", "#98508c", "#904e94"]
